The default debug formatter for Struct types.

Example

use bevy_reflect::Reflect;
#[derive(Reflect)]
struct MyStruct {
  foo: usize
}

let my_struct: &dyn Reflect = &MyStruct { foo: 123 };
println!("{:#?}", my_struct);

// Output:

// MyStruct {
//   foo: 123,
// }
